# isHex

### Usage

```javascript
import { isHex, isHexColor } from isHex; // ES6
// var IsHex = require('isHex'); // ES5

console.log(isHex("FFF")); // returns true
console.log(isHexColor("#FFFFFF")); // returns true
console.log(isHexColor("FFFFFF")); // returns true
console.log(isHex("FGF")); // returns false
```
